export const PickTotalFeeStep = (): JSX.Element => {
  const { course, siteSetting } = useEnrolState()
  const { t } = useTranslation()

  const [enrolForm, setEnrolForm] = useRecoilState(enrolState)

  const selectedSession =
    enrolForm.selectedClassData[enrolForm.currentSelectedClassIndex].selectedClass
  const selectedSessionDates =
    enrolForm.selectedClassData[enrolForm.currentSelectedClassIndex].selectedRegularPeriod

  useEffect(() => {
    if (!selectedSession || !selectedSessionDates) {
      setEnrolForm((prev) => ({ ...prev, currentStep: enrolForm.currentStep - 1 }))
      toast.warning(t('errors:ENROL.PLEASE_PICK_SESSION') as string)
    }
  }, [])

  if (!selectedSession || !selectedSessionDates) {
    return <></>
  }

  const singlePrice = getPriceWithCurrency(siteSetting?.currency, selectedSession.totalFee ?? 0)

  return (
    <div className="box-col items-start p-0">
      <Heading>{t('enrol:pickWorkshopFeeStep.pickOption')}</Heading>
      <Text>{t('enrol:pickWorkshopFeeStep.confirmDetail')}</Text>

      <BoxWithTextAction
        text={singlePrice}
        icon={<FaChevronRight />}
        action={() => {
          setEnrolForm((prev) => ({
            ...prev,
            currentStep: enrolForm.currentStep + 1,
            tuition: {
              originalFee: selectedSession.totalFee,
              paymentAmount: selectedSession.totalFee,
            },
          }))
        }}
      />
    </div>
  )
}

export const PickTimeslotStep = (): JSX.Element => {
  const { t } = useTranslation()
  const { course, originalUrl } = useEnrolState()
  const [enrolForm, setEnrolForm] = useRecoilState(enrolState)
  const today = new Date()
  const router = useRouter()
  const [currentTheme, setCurrentTheme] = useRecoilState(currentWebsiteTheme)

  const [prevSelectedOption, setPrevSelectedOption] = useRecoilState(prevSelectedOptionState)

  const { mutateAsync } = useMutation({
    mutationFn: (body: GetEnrolPriceData) => getEnrolPrice(body),
    onSuccess: (data: GetEnrolPriceResponse) => {
      if (data) {
        handleUpdateEnrolFormTuition(data)
        toast.success(t('enrol:payment.success') as string)
      } else {
        toast.error(t('enrol:payment.failure') as string)
      }
    },
    onError: (e) => {
      toast.error(t('enrol:payment.failure') as string)
    },
  })

  const sessionDates = (enrolForm.selectedSessionData as SelectedSessionDataState)?.selectedSession
    ?.sessionDates

  useEffect(() => {
    if (!course) {
      toast.error(t('errors:ENROL.PLEASE_PICK_SESSION') as string)
      if (!originalUrl || originalUrl === '') {
        router.push('/')
      } else {
        router.push(originalUrl)
      }
    }

    if (!enrolForm.selectedSessionData || !('selectedSession' in enrolForm.selectedSessionData)) {
      toast.warning(t('errors:ENROL.PLEASE_PICK_SESSION') as string)
      setEnrolForm((prev) => ({ ...prev, currentStep: enrolForm.currentStep - 1 }))
    }
  }, [course, sessionDates])

  if (!course || !sessionDates || sessionDates.length === 0) {
    return <Text>{t('enrol:pickPeriodStep.noSessions')}</Text>
  }

  const handleUpdateEnrolFormTuition = (data: GetEnrolPriceResponse) => {
    const newTuition = {
      originalFee: data.originalFee,
      paymentAmount: data.paymentAmount,
      currency: data.currency,
    }
    setPrevSelectedOption(enrolForm)
    setEnrolForm((prev) => ({ ...prev, tuition: newTuition }))
  }

  const sessionDateAfterToday = sessionDates.filter((session) => {
    const startTime = new Date(session.startTime)
    return startTime > new Date()
  })

  return (
    <div className="box-col w-full items-start p-0">
      <Heading>{t('enrol:pickPeriodStep.pickDateSession')}</Heading>
      <div className="font-bold" id="selectedSession">
        {enrolForm.selectedSessionData?.selectedSession?.name}
      </div>
      <div className="font-bold">{t('enrol:pickTimeslotStep.sessionRemind')}</div>

      {/* <Text>
        {t('enrol:pickPeriodStep.time')}
        {course.site.timeZone.id} {moment.tz(course.site.timeZone.id).format('Z')}
        {t('enrol:pickPeriodStep.pickDateSession')}{' '}
      </Text> */}
      {sessionDateAfterToday.length > 0 ? (
        <div className="flex w-full transform flex-col gap-2 rounded-lg border p-1">
          {sessionDateAfterToday.map((sessionDateItem, index) => {
            const start = moment(new Date(sessionDateItem.startTime)).format('YYYY/MM/DD hh:mm a')
            const end = moment(new Date(sessionDateItem.endTime)).format('YYYY/MM/DD hh:mm a')
            return (
              <BoxWithTextAction
                key={sessionDateItem.id}
                text={start + ' - ' + end}
                icon={<FaChevronRight />}
                action={() => {
                  // handlePickTuition({ scheduleItem, period: scheduleItem.period, lesson })
                  const newClassData = {
                    ...enrolForm.selectedSessionData,
                    selectedSessionDates: sessionDates,
                    selectedDate: sessionDateItem,
                  }
                  setPrevSelectedOption(enrolForm)
                  setEnrolForm((prev) => ({
                    ...prev,
                    currentStep: enrolForm.currentStep + 1,
                    selectedSessionData: newClassData,
                  }))
                  enrolCourseScrollAction(currentTheme)
                }}
                disabled={!!index}
              />
            )
          })}
        </div>
      ) : (
        <Text className="font-bold">{t('enrol:pickPeriodStep.noCurrentSessions')}</Text>
      )}
    </div>
  )
}

export const PickSessionStep = (): JSX.Element => {
  const { course, originalUrl } = useEnrolState()
  const [enrolForm, setEnrolForm] = useRecoilState(enrolState)
  const [sessions, setSessions] = useState(course?.sessions)
  const [currentTheme, setCurrentTheme] = useRecoilState(currentWebsiteTheme)
  const { t } = useTranslation()
  const [prevSelectedOption, setPrevSelectedOption] = useRecoilState(prevSelectedOptionState)
  const router = useRouter()

  const sortSessionsByNumberOfTimeslot = () => {
    const sessions = course?.sessions?.sort((sessionA, sessionB) => {
      const sessionDateAfterTodayA = sessionA.sessionDates.filter((session) => {
        const startTime = new Date(session.startTime)
        return startTime > new Date()
      })
      const sessionDateAfterTodayB = sessionB.sessionDates.filter((session) => {
        const startTime = new Date(session.startTime)
        return startTime > new Date()
      })

      return sessionDateAfterTodayB.length - sessionDateAfterTodayA.length
    })
    return sessions
  }

  const sortedSessions = useMemo(() => {
    if (!sessions) return []
    if (!course?.courseActivitiesOrder) {
      return sortSessionsByNumberOfTimeslot()
    }
    const { activityOrder } = course.courseActivitiesOrder
    const sortedWithOrder = rearrangeOrder(sessions, activityOrder)
    return sortedWithOrder
  }, [sessions, course?.courseActivitiesOrder])

  const getSortedSessionsByApplicationStatus = (
    status: 'applicable' | 'notApplicable'
  ): SessionWithEnrollCount[] => {
    const isApplicable = (session: Session) => {
      return status === 'applicable'
        ? session.sessionDates.every(({ startTime }) => new Date(startTime) > new Date())
        : session.sessionDates.some(({ startTime }) => new Date(startTime) <= new Date())
    }

    return (sortedSessions as SessionWithEnrollCount[])?.filter(isApplicable) ?? []
  }

  const SessionSelectionBox = ({ applicable }: { applicable: boolean }): JSX.Element => {
    return (
      <>
        {getSortedSessionsByApplicationStatus(applicable ? 'applicable' : 'notApplicable').map(
          (sessionItem) => {
            return (
              <div key={sessionItem.id} className="box-col">
                <BoxWithTextAction
                  percentage={
                    applicable ? (sessionItem.enrollCount / sessionItem.quota) * 100 : undefined
                  }
                  text={sessionItem.name}
                  icon={<FaChevronRight />}
                  action={() => {
                    const newClassData = {
                      ...enrolForm.selectedSessionData,
                      selectedSession: sessionItem,
                    }
                    setPrevSelectedOption(enrolForm)
                    setEnrolForm((prev) => ({
                      ...prev,
                      currentStep: enrolForm.currentStep + 1,
                      selectedSessionData: newClassData,
                    }))
                    enrolCourseScrollAction(currentTheme)
                  }}
                  disabled={!applicable || sessionItem.enrollCount >= sessionItem.quota}
                />
              </div>
            )
          }
        )}
      </>
    )
  }

  useEffect(() => {
    if (!course) {
      toast.error(t('errors:ENROL.PLEASE_PICK_SESSION') as string)
      if (!originalUrl || originalUrl === '') {
        router.push('/')
      } else {
        router.push(originalUrl)
      }
    }

    if (!sessions || sessions.length === 0) {
      toast.error(t('errors:ENROL.WORKSHOP_NOT_OPEN') as string)
      setEnrolForm((prev) => ({ ...prev, currentStep: enrolForm.currentStep - 1 }))
    }
  }, [])

  if (!course || !sessions || sessions.length === 0) {
    return <Text>{t('enrol:pickPeriodStep.noSessions')}</Text>
  }

  return (
    <div className="box-col items-start p-0">
      <Heading>{t('enrol:workshop.pickSession')}</Heading>
      <SessionSelectionBox applicable />
      {getSortedSessionsByApplicationStatus('notApplicable').length > 0 && (
        <Text className="mt-4">{t('enrol:pickPeriodStep.statedSession')}</Text>
      )}
      <SessionSelectionBox applicable={false} />
    </div>
  )
}
export default PickSessionStep
